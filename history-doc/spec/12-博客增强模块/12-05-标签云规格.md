## 12.5.2 具体规格

### 12.5.2.1 组件结构

```vue
<!-- components/blog/TagCloud.vue -->
<template>
  <div class="tag-cloud" v-if="tags.length > 0">
    <span
      v-for="tag in tags"
      :key="tag.id"
      class="tag-item"
      :style="{ fontSize: calculateFontSize(tag.postCount) + 'px' }"
      :title="`${tag.postCount} 篇文章`"
      @click="handleClick(tag)"
    >
      {{ tag.name }}
    </span>
  </div>
  <div v-else class="no-tags">暂无标签</div>
</template>
```

### 12.5.2.2 字体大小计算

```typescript
function calculateFontSize(postCount: number): number {
  const minSize = 12
  const maxSize = 28
  const maxPostCount = Math.max(...tags.value.map(t => t.postCount || 0), 1)

  return minSize + (postCount / maxPostCount) * (maxSize - minSize)
}
```

```typescript
// 移动端显示热门标签（前10个）
const displayTags = computed(() => {
  if (isMobile.value) {
    return tags.value
      .sort((a, b) => (b.postCount || 0) - (a.postCount || 0))
      .slice(0, 10)
  }
  return tags.value
})
```

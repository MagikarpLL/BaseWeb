## 命名规范

### 组件命名

| 类型 | 规范 | 示例 |
|------|------|------|
| 组件文件 | PascalCase | `AppHeader.vue` |
| 组件名 | PascalCase | `name: 'AppHeader'` |
| 组合式函数 | camelCase + use 前缀 | `useAuth.ts` |
| 工具函数 | camelCase | `formatDate.ts` |

### 目录/文件命名

| 类型 | 规范 | 示例 |
|------|------|------|
| 目录 | kebab-case | `blog-post`, `public-api` |
| Vue 文件 | PascalCase | `HomePage.vue` |
| TS/JS 文件 | camelCase | `authApi.ts` |
| 样式文件 | kebab-case | `global.scss` |

### 变量命名

| 类型 | 规范 | 示例 |
|------|------|------|
| 普通变量 | camelCase | `postList`, `isLoading` |
| 常量 | UPPER_SNAKE_CASE | `API_BASE_URL` |
| 响应式状态 | 以 `ref` 或 `reactive` 结尾 | `postListRef`, `formState` |
| 计算属性 | 描述性名词 | `formattedDate`, `totalPages` |
| 事件处理 | handle 前缀 | `handleSubmit`, `handleClick` |

## Vue SFC 规范

### `<script setup>` 结构顺序

```vue
<script setup lang="ts">
// 1. 导入 (Imports)
import { ref, computed, onMounted } from 'vue'
import { useRoute } from 'vue-router'
import { useAuthStore } from '@/stores/auth'

// 2. Props / Emits
const props = defineProps<{ postId: number }>()
const emit = defineEmits<{ (e: 'update', value: string): void }>()

// 3. 响应式状态 (Reactive State)
const isLoading = ref(false)
const post = ref<Post | null>(null)

// 4. 计算属性 (Computed)
const formattedDate = computed(() => {
  return post.value?.publishedAt.format('YYYY-MM-DD')
})

// 5. 方法 (Methods)
async function fetchPost() {
  isLoading.value = true
  try {
    post.value = await postApi.getPost(props.postId)
  } finally {
    isLoading.value = false
  }
}

// 6. 生命周期 (Lifecycle)
onMounted(fetchPost)

// 7. 监听器 (Watch) - 如有
</script>

<template>
  <!-- 模板内容 -->
</template>

<style scoped lang="scss">
/* 样式内容 */
</style>
```

## API 调用规范

### Axios 实例配置

```typescript
// api/axios.ts
const api = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL,
  timeout: 30000,
  withCredentials: true,  // 发送 Cookie
})

// 请求拦截器: 附加 Token
api.interceptors.request.use((config) => {
  const authStore = useAuthStore()
  if (authStore.accessToken) {
    config.headers.Authorization = `Bearer ${authStore.accessToken}`
  }
  return config
})

// 响应拦截器: 处理 401 刷新
api.interceptors.response.use(...)
```

### API 模块封装

```typescript
// api/blog/postApi.ts
export const postApi = {
  // 获取列表
  getPosts(params: PostQueryParams) {
    return api.get<ApiResponse<PaginatedResponse<Post>>>('/blog/posts', { params })
  },

  // 获取详情
  getPost(slug: string) {
    return api.get<ApiResponse<PostDetail>>(`/blog/posts/${slug}`)
  },

  // 创建
  createPost(data: CreatePostRequest) {
    return api.post<ApiResponse<Post>>('/admin/posts', data)
  },

  // 更新
  updatePost(id: number, data: UpdatePostRequest) {
    return api.put<ApiResponse<Post>>(`/admin/posts/${id}`, data)
  },

  // 删除
  deletePost(id: number) {
    return api.delete<ApiResponse<void>>(`/admin/posts/${id}`)
  },
}
```

## 状态管理规范

### Store 定义

```typescript
// stores/auth.ts
export const useAuthStore = defineStore('auth', () => {
  // 状态
  const accessToken = ref<string | null>(null)
  const user = ref<User | null>(null)

  // Getter
  const isAuthenticated = computed(() => !!accessToken.value)
  const isAdmin = computed(() => user.value?.role === 'ADMIN')

  // Action
  function setAuth(token: string, userData: User) {
    accessToken.value = token
    user.value = userData
  }

  function clearAuth() {
    accessToken.value = null
    user.value = null
  }

  return {
    accessToken,
    user,
    isAuthenticated,
    isAdmin,
    setAuth,
    clearAuth,
  }
})
```

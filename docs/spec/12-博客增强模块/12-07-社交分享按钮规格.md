## 12.7.2 具体规格

### 12.7.2.1 组件结构

```vue

```

### 12.7.2.2 分享函数实现

```typescript
const currentUrl = computed(() => window.location.href)

function shareToTwitter() {
  const url = `https://twitter.com/intent/tweet?text=${encodeURIComponent(props.title)}&url=${encodeURIComponent(currentUrl.value)}`
  window.open(url, '_blank', 'width=550,height=420')
}

function shareToWeibo() {
  const url = `https://service.weibo.com/share/share.php?url=${encodeURIComponent(currentUrl.value)}&title=${encodeURIComponent(props.title)}`
  window.open(url, '_blank', 'width=550,height=420')
}

function shareToQQ() {
  const url = `https://connect.qq.com/widget/share/iframe/choose.html?url=${encodeURIComponent(currentUrl.value)}&title=${encodeURIComponent(props.title)}`
  window.open(url, '_blank', 'width=550,height=420')
}
```

### 12.7.2.3 微信二维码生成

使用 `qrcode` 库生成二维码：

```typescript
import QRCode from 'qrcode'

async function showWechatQR() {
  showQR.value = true
  await nextTick()
  if (qrCanvas.value) {
    QRCode.toCanvas(qrCanvas.value, currentUrl.value, {
      width: 200,
      margin: 2
    })
  }
}
```

### 12.7.2.4 复制链接

```typescript
async function copyLink() {
  try {
    await navigator.clipboard.writeText(currentUrl.value)
    ElMessage.success('已复制到剪贴板')
  } catch {
    // 降级处理
    const textarea = document.createElement('textarea')
    textarea.value = currentUrl.value
    document.body.appendChild(textarea)
    textarea.select()
    document.execCommand('copy')
    document.body.removeChild(textarea)
    ElMessage.success('已复制到剪贴板')
  }
}
```

### 12.7.2.5 移动端 Web Share API

```typescript
async function mobileShare() {
  if (navigator.share) {
    try {
      await navigator.share({
        title: props.title,
        text: props.excerpt,
        url: currentUrl.value
      })
    } catch (err) {
      // 用户取消分享，不提示错误
    }
  }
}
```
